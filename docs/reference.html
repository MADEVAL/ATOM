Database

use Atom\Database\Database;

$db = new Database($config->get('DB_DSN'));
// SQLite: 'sqlite:/path/to/db.sqlite'
// MySQL:  'mysql:host=localhost;dbname=app;charset=utf8mb4', user, pass

$db->all('SELECT * FROM users WHERE active = ?', [1]);
$db->one('SELECT * FROM users WHERE id = ?', [$id]);
$db->single('SELECT COUNT(*) FROM users');
$db->run('UPDATE users SET name = ? WHERE id = ?', [...] );
$db->lastId();

// Transactions
$db->beginTransaction();
$db->commit();
$db->rollback();

Logger

use Atom\Support\Logger;

$log = new Logger('/var/log/app.log');
$log->debug('query', ['sql' => $sql, 'ms' => 12.5]);
$log->info('login', ['user' => 42]);
$log->warn('rate limit');
$log->error('payment failed');
$log->critical('disk full');
$log->alert('db down');
$log->emergency('outage');

// LEVELS: DEBUG(0) INFO(1) WARN(2) ERROR(3) CRITICAL(4) ALERT(5) EMERGENCY(6)
$log = new Logger('/var/log/app.log', Logger::WARN);

// Rotation & clear
$log->rotate(); // rename → app_20260101_120000.log
$log->clear();  // delete file entirely

// Auto‑rotation by size
$log = new Logger('/var/log/app.log', maxSize: 1_048_576);

Paginator

use Atom\Support\Paginator;

$p = Paginator::from($req, defaultPerPage: 20);

// SQL helper
$sql = "SELECT * FROM posts " . $p->sqlLimit();
// → LIMIT 20 OFFSET 0

// Wrap results
$result = $p->paginate($items, $totalCount);
// → ['data' => [...], 'page' => 1, 'perPage' => 20, 'total' => 150, 'pages' => 8]

// Query params: ?page=2&per_page=10
// page clamped ≥ 1, per_page clamped 1..100

Encryption

use Atom\Support\Encrypt;

$encrypted = Encrypt::encrypt('sensitive data', $app->config->get('APP_KEY'));
$plain     = Encrypt::decrypt($encrypted, $app->config->get('APP_KEY'));

// AES‑256‑GCM with random IV each encryption
// Wrong key or tampered payload → RuntimeException

CLI Console

php bin/atom list      # show commands with descriptions
php bin/atom help      # alias for list
php bin/atom routes    # list registered routes
php bin/atom cache     # clear compiled cache
use Atom\Console\Console;

$console = new Console($app);
$console->add('db:migrate', function(array $args, array $options): int { ...
    return 0;
}, 'Run database migrations');
$console->run($argv);

// Options via PCRE: --flag, --key=value
php bin/atom db:migrate --force --env=production

// NO_COLOR support
NO_COLOR=1 php bin/atom routes       # environment variable
php bin/atom routes --no-color        # command-line flag

.env & Config

// .env file
APP_DEBUG=true
APP_NAME=MyApp
APP_ENV=production
APP_CACHE_DIR=/tmp/cache
APP_VIEWS_DIR=/var/views
APP_TIMEZONE=Europe/Moscow
APP_LOG_FILE=/var/log/app.log
APP_LOG_LEVEL=WARN
APP_LOG_MAX_SIZE=1048576
DB_DSN=mysql:host=localhost;dbname=app

// .env.production overrides .env (loaded when APP_ENV=production)
APP_DEBUG=false
APP_LOG_LEVEL=ERROR

// Load
$config = Config::fromEnv(__DIR__ . '/.env');
$config->debug;             // false (overridden)
$config->appName;           // MyApp
$config->timezone;          // Europe/Moscow
$config->logLevel;          // 3 (ERROR)
$config->get('DB_DSN');     // mysql:host=...
$config->get('MISSING', 'default'); // default

Regex utility

use Atom\Support\Regex;

Regex::match('#(\d+)#', 'abc123');          // → ['123'] | null
Regex::matchAll('#(\w+)#', 'a b c');       // → [['a','b','c']]
Regex::replace('#\s+#', '-', 'a b');      // → 'a-b'
Regex::split('#,#', 'a,b,c');              // → ['a','b','c']
Regex::quote('foo.bar');                    // → 'foo\.bar'
Regex::assert('#valid_pattern#');           // throws on bad regex
