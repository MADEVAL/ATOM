Database

use Atom\Database\Database;

$db = new Database($config->get('DB_DSN'));
// SQLite: 'sqlite:/path/to/db.sqlite'
// MySQL:  'mysql:host=localhost;dbname=app;charset=utf8mb4', user, pass

$db->all('SELECT * FROM users WHERE active = ?', [1]);
$db->one('SELECT * FROM users WHERE id = ?', [$id]);
$db->single('SELECT COUNT(*) FROM users');
$db->run('UPDATE users SET name = ? WHERE id = ?', [...] );
$db->lastId();

// Transactions
$db->beginTransaction();
$db->commit();
$db->rollback();

Logger

use Atom\Support\Logger;

$log = $app->log();  // obtain via Application (respects config)
$log = new Logger('/var/log/app.log');
$log->debug('query', ['sql' => $sql, 'ms' => 12.5]);
$log->info('login', ['user' => 42]);
$log->warn('rate limit');
$log->error('payment failed');
$log->critical('disk full');
$log->alert('db down');
$log->emergency('outage');

// LEVELS: DEBUG(0) INFO(1) WARN(2) ERROR(3) CRITICAL(4) ALERT(5) EMERGENCY(6)
$log = new Logger('/var/log/app.log', Logger::WARN);

// Rotation & clear
$log->rotate(); // rename → app_20260101_120000.log
$log->clear();  // delete file entirely

// Auto‑rotation by size
$log = new Logger('/var/log/app.log', maxSize: 1_048_576);

Cache

use Atom\Cache\{Cache, ArrayDriver, FileDriver};

// Obtain via Application (lazy, config-driven)
$cache = $app->cache();

// Basic operations
$cache->set('key', 'value', ttl: 3600);
$value = $cache->get('key', 'default');
$exists = $cache->has('key');
$cache->delete('key');
$cache->flush();

// Remember — fetch from closure if missing
$data = $cache->remember('stats', fn() => heavyQuery(), ttl: 600);

// Remember forever (no TTL)
$data = $cache->rememberForever('config', fn() => loadConfig());

// Increment / decrement counters
$views = $cache->increment('page_views');
$stock = $cache->decrement('stock_count', 2);

// Driver access
$driver = $cache->driver(); // → Driver instance

The Driver interface defines the contract: get(), set(key, value, ttl), has(), delete(), flush(). Two drivers ship with Atom:

  • ArrayDriver — in‑memory, per‑request. TTL honoured via timestamp check.
  • FileDriver — persisted to disk (cacheDir/sha1(key).php). TTL stored in first line, atomic writes via temp‑file + rename. Probabilistic garbage collection via cleanupChance.

Set APP_CACHE_DRIVER=array for array driver, omit (or =file) for file driver.

Paginator

use Atom\Support\Paginator;

// From Request (reads ?page= & &per_page=)
$p = Paginator::from($req, defaultPerPage: 20);

// Or manual construction
$p = Paginator::make(page: $page, perPage: $perPage);

// SQL helper
$sql = "SELECT * FROM posts " . $p->sqlLimit();
// → LIMIT 20 OFFSET 0

// Wrap results
$result = $p->paginate($items, $totalCount);
// → ['data' => [...], 'page' => 1, 'perPage' => 20, 'total' => 150, 'pages' => 8]

// Query params: ?page=2&per_page=10
// page clamped ≥ 1, per_page clamped 1..100

ORM

use Atom\Orm\{Model, Table, PrimaryKey, Column};
use Atom\Database\Database;

// Set global connection
Model::setConnection($db);

// Define a model
#[Table('users')]
class User extends Model {
    #[PrimaryKey] public int $id = 0;
    #[Column('user_name')] public string $name = '';
    public string $email = '';
}

// CRUD
$user = User::create(['name' => 'Alice', 'email' => 'a@ex.com']);
$user = User::find(1);
$user = User::findOrFail(42);
$user->name = 'Bob';
$user->save();
$user->delete();
User::destroy(1, 2, 3);

// Query builder
$users = User::query()->where('active', 1)->whereIn('role', ['admin','user'])->orderBy('created_at', 'DESC')->limit(10)->get();
$total = User::query()->where('active', 1)->count();
$exists = User::query()->where('email', 'a@ex.com')->exists();

// Magic whereX methods
$admin = User::query()->whereRole('admin')->get();
$active = User::query()->whereStatus('active')->get();

// Relations
class Post extends Model {
    public function author() { return $this->belongsTo(User::class); }
    public function comments() { return $this->hasMany(Comment::class); }
    public function meta() { return $this->hasOne(PostMeta::class); }
}

// Eager loading
$posts = Post::query()->with('author', 'comments')->get();
foreach ($posts as $post) {
    $post->author()->getResults()->name;     // user model
    foreach ($post->comments()->getResults() as $c) echo $c->body;
}

// Paginate via query builder
$pager = Post::query()->where('status', 'published')->paginate(20);

// First or create
$user = User::firstOrCreate(['email' => 'a@ex.com'], ['name' => 'Alice']);

// Fill from array
$user = new User();
$user->fill(['name' => 'Bob', 'email' => 'b@ex.com']);
$user->save();

Timestamps: Models with created_at / updated_at public properties automatically receive Y-m-d H:i:s values on insert/update. Set protected bool $timestamps = false to disable.

Type casting: Public property types (int, float, bool, string) are auto‑cast on fill and hydration. #[Table] sets the DB table name (auto‑derived from class name otherwise). #[PrimaryKey] marks the PK column. #[Column('name')] maps a property name to a different column name.

Encryption

use Atom\Support\Encrypt;

$encrypted = Encrypt::encrypt('sensitive data', $app->config->get('APP_KEY'));
$plain     = Encrypt::decrypt($encrypted, $app->config->get('APP_KEY'));

// AES‑256‑GCM with random IV each encryption
// Wrong key or tampered payload → RuntimeException

CLI Console

php bin/atom cache     # clear compiled cache
php bin/atom ws:serve  # start WebSocket server (--host, --port)

$console = new Console($app);
$console->add('db:migrate', function(array $args, array $options): int { ...
    return 0;
}, 'Run database migrations');
$console->run($argv);

// Options via PCRE: --flag, --key=value
php bin/atom db:migrate --force --env=production

// NO_COLOR support

Regex utility

use Atom\Support\Regex;

Regex::match('#(\d+)#', 'abc123');          // → ['123'] | null
Regex::matchAll('#(\w+)#', 'a b c');       // → [['a','b','c']]
Regex::replace('#\s+#', '-', 'a b');      // → 'a-b'
Regex::split('#,#', 'a,b,c');              // → ['a','b','c']
Regex::quote('foo.bar');                    // → 'foo\.bar'
Regex::assert('#valid_pattern#');           // throws on bad regex
