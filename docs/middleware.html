Middleware pipeline

class Auth implements MiddlewareInterface {
    public function handle(Request $req, Closure $next): Response {
        if (!$req->bearer) {
            return new Response('Unauthorized', StatusCode::UNAUTHORIZED);
        }
        return $next($req);
    }
}

// Object, string (DI-resolved), or Closure
$r->get('/secure', 'Controller@action', '', ['Auth']);
$r->get('/debug', 'DebugController@index', '', [
    fn(Request $req, Closure $next) => $next($req)->withHeader('X-Debug', '1')
]);

CORS middleware

use Atom\Middleware\Cors;

// Default: allow *, all methods, Content-Type + Authorization
$app->router->group('/api', [Cors::class], fn($r) => { ... });

$app->router->group('/api', [new Cors(
    allowOrigin: 'https://example.com',
    allowMethods: 'GET,POST,PATCH',
    allowHeaders: 'X-API-Key,Content-Type',
    allowCredentials: true,
    exposeHeaders: 'X-Total-Count',
)], fn($r) => { ... });

Handles OPTIONS preflight. Reflects Origin when allowOrigin: '*'. * + credentials throws InvalidArgumentException per CORS spec.

CSRF middleware

use Atom\Middleware\Csrf;

$app->router->group('', [Csrf::class], fn($r) => { ... });

// Template: <input type="hidden" name="_csrf" value="{{ _csrf }}">
// Or header: X-CSRF-Token: <token>
// Validated on POST/PUT/PATCH/DELETE - 403 on mismatch
// Token rotated after successful validation

Rate Limiter

use Atom\Middleware\RateLimit;

$app->router->group('/api', [new RateLimit(max: 100, window: 60)], fn($r) => { ... });

Per‑IP‑and‑path tracking. Exceeded limit → 429 Too Many Requests.
